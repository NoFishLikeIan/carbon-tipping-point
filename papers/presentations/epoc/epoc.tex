\section{Social Planner Solution}
\begin{frame}{Social Planner Problem}
    \begin{columns}
        \column{0.4\linewidth}
        \visible<1->{
            \textbf{State variables}
            \begin{enumerate}
                \item Temperature $\T$
                \item CO$_2$ conc. $\M$
                \item GDP $\Y$
            \end{enumerate}
        }
        \visible<2->{
            \textbf{Control variables}
            \begin{enumerate}
                \item Consumption r. $\control{\chi}$
                \item Abatement r. $\control{\alpha}$
            \end{enumerate}
        }
        \column{0.6\linewidth}
        \visible<3->{
            \textbf{Objective}
            \begin{equation*}
                v_t(\T, \M, \Y) = \sup_{\control{\chi, \alpha}} \mathbb{E}_t \int^\infty_t f(\control{\chi}_s Y_s, v_s) \; \dd{s} 
            \end{equation*}
            \begin{itemize}
                \item Discount rate $\rho = 1.5 \%$
                \item RRA $\theta = 2$
                \item EIS $\psi = 1.5$
            \end{itemize}
        }
    \end{columns}
\end{frame}

\begin{frame}{Numerical Solution}
    \begin{enumerate}
        \item Finite Difference scheme not robust: monotonicity of scheme
        \item Develop a Markov Chain approximation for Epstein-Zin specification based on \cite{kushner_numerical_2001}
    \end{enumerate}
\end{frame}

\begin{frame}{Benchmark Model: Stochastic Tipping}
    \begin{equation*}
        \mu_{\lambda}(\T, \M) = {\color{cbyellow} S_0} \; (1 - \cancelto{\bar{\lambda}}{\lambda(\T)})
        - \eta \sigma \T^4 + (G_0 + G_1 \log \M / M^{\mathrm{p}})
    \end{equation*}
    \begin{equation*}
        \epsilon \; \dd{{\T} = \mu_{\lambda}(\T, \M) + \sigma \dd{w} + \Theta(\T) \; \dd{N} \text{ with arrival } \pi(\T)
    \end{equation*}
    Calibrated as in \cite{hambel_optimal_2021}
\end{frame}
