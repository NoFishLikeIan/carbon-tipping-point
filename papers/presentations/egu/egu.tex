\section{Climate Model}
\begin{frame}
    Textbook energy balance \citeauthor{ashwin_tipping_2012} (\citeyear{ashwin_tipping_2012}) model for temperature $\T$ and CO$_2$ concentration $\M$ dynamics \vspace{5pt}
    \only<1>{\begin{equation}
        \epsilon \; \text{d} \T = \mu(\T, \M) \; \dd{t} + \sigma \; \dd{W} + \epsilon(\T) \; \dd{N}
    \end{equation} \vspace{2pt}
    \begin{equation}
        \mu(\T, \M) = S + G_0 + G_1 \log(\M / M^p) - \eta \sigma \T^4
    \end{equation}}
    \only<2>{
        \begin{equation}
            \epsilon \; \text{d} \T = \mu(\T, \M) \; \dd{t} + \sigma \; \dd{W} + {\color{red} \cancel{{\color{black} \epsilon(\T) \; \dd{N}}}}
        \end{equation} \vspace{2pt}
        \begin{equation}
            \mu(\T, \M) = S \underbrace{(1 - \lambda(\T))}_{\text{Ice-albedo}} + G_0 + G_1 \log(\M / M^p) - \eta \sigma \T^4
        \end{equation}
    }
\end{frame}

\section{Economy}
\begin{frame}
    Maximise NPV of consumption subject to\vspace{2em}
    \begin{equation*}
        \frac{\dd{{\Y}}{\Y} = \Big(\text{tech. growth} - \text{abatement} - \text{damages}(\T)\Big)\; \dd{t} + \sigma \dd{W}
    \end{equation*} \vspace{2em}
    \begin{equation*}
        \frac{\dd{{\M}}{\M} = \Big(\text{BAU} - \text{abatement}\Big)\; \dd{t}
    \end{equation*}
\end{frame}
